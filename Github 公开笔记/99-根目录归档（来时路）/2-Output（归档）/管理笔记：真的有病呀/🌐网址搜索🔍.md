const initSearch = () => {
    const file = app.workspace.getActiveFile();
    const targetFilePath = file?.path;
    if (!file) {
        throw new Error('No active file found');
    }

    const searchTermTarget = mb.createBindTarget('frontmatter', file.path, ['searchTerm']);

    const searchEngines = [
        { name: 'Baidu', url: 'https://www.baidu.com/s?wd=%s' },
        { name: 'Google', url: 'https://www.google.com/search?q=%s' },
        { name: 'metaAll', url: 'https://metaso.cn/?q=%s' },
        { name: 'metaScholar', url: 'https://metaso.cn/?q=%s&m=detail&et=scholar' },
        { name: 'metaPdf', url: 'https://metaso.cn/?q=%s&m=detail&et=pdf' },
        { name: 'metaImage', url: 'https://metaso.cn/?q=%s&m=detail&et=image' },
        { name: 'metaPodcast', url: 'https://metaso.cn/?q=%s&m=detail&et=podcast' },
        { name: 'redBook', url: 'https://www.xiaohongshu.com/search_result?keyword=%s' },
        { name: 'Bilibili', url: 'https://search.bilibili.com/all?&keyword=%s' },
    ];

    const engineBindTargets = searchEngines.map(engine => ({
        ...engine,
        bindTarget: mb.createBindTarget('frontmatter', file.path, [engine.name])
    }));

    app.metadataCache.on('changed', (changedFile) => {
        if (changedFile.path === targetFilePath) {
            const frontmatter = app.metadataCache.getFileCache(file)?.frontmatter;
            if (frontmatter && frontmatter.searchTerm) {
                const searchTerm = frontmatter.searchTerm;
                
                engineBindTargets.forEach(engine => {
                    const link = engine.url.replace('%s', encodeURIComponent(searchTerm));
                    mb.updateMetadata(engine.bindTarget, () => link);
                });
            }
        }
    });
};
